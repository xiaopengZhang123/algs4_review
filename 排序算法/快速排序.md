# partition过程
快速排序最重要的部分，或者说最有意思的部分就是这个partition的过程
```
int partition(int[]arr,int l,int r){
    int pivot=arr[l];
    int j=l;
    for(int i=l+1;i<=r;i++){
        if(arr[i]<=pivot){
            j++;
            swap(arr,i,j);
        }
    }
    swap(arr,j,l);
    return j;
}

```

```
另一种快排
void sort(int[]arr,int l,int r){
    if(l>=r){
        return;
    }
    int i=l-1,j=r+1,pivot=arr[l+r>>1];
    while(i<j){
        do i++;
        while(arr[i]<pivot);
        do j--;
        while(arr[i]>pivot);
        if(i<j){
            swap(arr,i,j);
        }
    }
    sort(arr,l,j);
    sort(arr,j+1,r);
}

```
# 三路快排
```
void sort(int[]arr,int l,int r){
    if(l>=r){
        return;
    }
    int lt=l,gt=r,pivot=arr[l+r>>1];
    int i=l,k=r;
    while(i<=gt){
        if(arr[i]<pivot){
            swap(arr,i,lt);
            lt++;
            //这里不加i++也可以的原因交换了轴和i元素，是走了arr[i]=pivot,实际上多比较了一次
            i++;
        }else if(arr[i]>pivot){
            swap(arr,i,gt);
            gt--;
        }else{
            i++;
        }
    }
    sort(arr,l,lt-1);
    sort(arr,gt+1,r);
}
```
